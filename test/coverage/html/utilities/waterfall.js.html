function waterfall(parent,className){
 
    const Parent=document.getElementById(parent);
    const postChild=Parent.getElementsByClassName(className);
    const postWidth=postChild[0].offsetWidth;
 
    const col=Math.floor((document.documentElement.clientWidth || document.body.clientWidth)/postWidth);
    Parent.style.cssText='width:'+postWidth*col+'px;margin:0 auto;';
    const colHeight=[];
    for(let i=0,len = postChild.length;i<len;i++){
 
        const postHeight=postChild[i].offsetHeight;
        if(i<col){
            colHeight.push(postHeight);
        }else{
            const minHeight=Math.min(...colHeight);
            const minHeightIndex=getMinHeightIndex(colHeight,minHeight);
            postChild[i].style.position='absolute';
            postChild[i].style.top=minHeight+'px';
            postChild[i].style.left=postChild[minHeightIndex].offsetLeft+'px';
            colHeight[minHeightIndex]+=postChild[i].offsetHeight;
        }
    }
}
 
function getMinHeightIndex(arr,min){
 
    for(let i = 0,len = arr.length;i<len;i++){
        if(arr[i] === min){
            return i
        }
    }
}
 
 
 
 
export  default waterfall;
